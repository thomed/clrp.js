clrp.js

A standalone library for color picker inputs

Installation

Code:
<script src="clrp.js">
<input class="clrp">
Result:

Basic Usage

Code:
<p id="my-text">Color:</p>
<input class="clrp" id="my-clrp" value="#5500cc">
<script>
    var myClrp = document.getElementById("my-clrp");
    var text = document.getElementById("my-text");
    
    myClrp.addEventListener('change', function() {
        text.innerText = "Color: " + myClrp.value;
        text.style.color = myClrp.value;
    });
</script>
